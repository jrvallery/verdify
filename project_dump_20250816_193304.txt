    """
    Request ID middleware and utilities for Project Verdify API.

    Provides X-Request-Id header injection for request traceability and correlation.
    """

    import uuid
    from typing import Callable

    from fastapi import Request, Response
    from starlette.middleware.base import BaseHTTPMiddleware


    class RequestIdMiddleware(BaseHTTPMiddleware):
        """
        Middleware that ensures every request has a unique X-Request-Id header.

        - If the client provides X-Request-Id, it's preserved for correlation
        - If not provided, a new UUID4 is generated
        - The request ID is always included in the response headers
        - The request ID is attached to the request state for access in handlers
        """

        async def dispatch(self, request: Request, call_next: Callable) -> Response:
            # Get existing request ID from headers or generate a new one
            request_id = request.headers.get("X-Request-Id")
            if not request_id:
                request_id = str(uuid.uuid4())

            # Store request ID in request state for access in error handlers and routes
            request.state.request_id = request_id

            # Process the request
            response = await call_next(request)

            # Check if this is a 404 response and convert to standardized format
            if response.status_code == 404 and response.headers.get("content-type") == "application/json":
                from app.utils_errors import create_error_response
                return create_error_response(
                    request=request,
                    status_code=404,
                    message="Not Found",
                    details="The requested resource was not found"
                )

            # Always include X-Request-Id in response headers for traceability
            response.headers["X-Request-Id"] = request_id

            return response


    def get_request_id(request: Request) -> str:
        """
        Extract request ID from request state.

        Returns:
            The request ID string, or a fallback UUID if not found
        """
        return getattr(request.state, "request_id", str(uuid.uuid4()))
